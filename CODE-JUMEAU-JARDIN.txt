  createDigitalTwinInGarden: (plantDefId, x, y, scanData) => {
    const state = get();
    
    // 1. Vérifier que la plante existe
    const plantDef = PLANTS[plantDefId];
    if (!plantDef) {
      return {
        success: false,
        message: `❌ Type de plante "${plantDefId}" non reconnu.`,
        rewards: null
      };
    }
    
    // 2. Vérifier l'espace du jardin
    const spacing = PLANT_SPACING[plantDefId];
    if (!spacing) {
      return {
        success: false,
        message: `❌ Pas d'espacement défini pour ${plantDef.name}.`,
        rewards: null
      };
    }
    
    // 3. Vérifier limites jardin
    if (x < 0 || y < 0 || x + spacing.plantSpacingCm > state.gardenWidthCm || y + spacing.rowSpacingCm > state.gardenHeightCm) {
      return {
        success: false,
        message: `❌ Position hors limites du jardin.\n\nJardin: ${state.gardenWidthCm}×${state.gardenHeightCm}cm`,
        rewards: null
      };
    }
    
    // 4. Vérifier chevauchement avec autres plants
    const overlaps = state.gardenPlants.some((gp) => {
      const s = PLANT_SPACING[gp.plantDefId];
      if (!s) return false;
      return x < gp.x + s.plantSpacingCm && x + spacing.plantSpacingCm > gp.x &&
             y < gp.y + s.rowSpacingCm && y + spacing.rowSpacingCm > gp.y;
    });
    
    if (overlaps) {
      return {
        success: false,
        message: `❌ Un autre plant occupe déjà cet emplacement.\n\nChoisissez une autre position.`,
        rewards: null
      };
    }
    
    // 5. Créer le plant avec l'état scanné
    const growthStage = scanData.growthStage || { stage: 2, estimatedAge: 15 };
    const healthStatus = scanData.healthStatus || { isHealthy: true, diseaseName: 'Sain' };
    
    const newPlant: PlantState = {
      ...createInitialPlantState(plantDefId),
      stage: Math.min(5, growthStage.stage),
      daysSincePlanting: growthStage.estimatedAge,
      daysInCurrentStage: Math.min(10, growthStage.estimatedAge),
      health: healthStatus.isHealthy ? 85 : 60,
      hasDisease: !healthStatus.isHealthy,
      diseaseDays: healthStatus.isHealthy ? 0 : 3,
      waterLevel: 70,
      fertilizerLevel: 50,
    };
    
    // 6. Ajouter au jardin
    const newGardenPlant: GardenPlant = {
      id: uid(),
      plantDefId,
      x: Math.round(x),
      y: Math.round(y),
      plant: newPlant,
    };
    
    const newGardenPlants = [...state.gardenPlants, newGardenPlant];
    saveGardenPlants(newGardenPlants);
    
    // 7. Récompenses
    const rewards = {
      coins: 50,
      xp: 100,
      bonusSeeds: null as { plantDefId: string; count: number } | null,
    };
    
    // Bonus qualité scan
    if (scanData.confidence > 0.8) {
      rewards.coins += 25;
      rewards.xp += 50;
    }
    
    // Bonus graines si stade avancé
    if (growthStage.stage >= 4) {
      rewards.bonusSeeds = { plantDefId, count: 3 };
      const newCollection = { ...state.seedCollection };
      newCollection[plantDefId] = (newCollection[plantDefId] || 0) + 3;
      saveSeedCollection(newCollection);
      set({ seedCollection: newCollection });
    }
    
    // 8. Pièces
    const newCoins = state.coins + rewards.coins;
    saveCoins(newCoins);
    
    // 9. Alert
    const alerts = [
      ...state.alerts.slice(-25),
      {
        id: `twin-${Date.now()}`,
        type: "success" as const,
        message: `✨ ${plantDef.emoji} ${plantDef.name} ajouté au jardin !`,
        emoji: "🌱",
        cellX: 0,
        cellY: 0,
        timestamp: Date.now(),
        severity: "info" as const,
      },
    ];
    
    set({
      gardenPlants: newGardenPlants,
      coins: newCoins,
      alerts,
    });
    
    useSoundManager.getState().playEventSound('plant');
    triggerVisualEffect("success-pop");
    
    return {
      success: true,
      message: `✅ Jumeau numérique planté !

🌱 ${plantDef.emoji} ${plantDef.name}
📍 Position: ${Math.round(x)}cm × ${Math.round(y)}cm
📊 Stade ${growthStage.stage}/5 (${growthStage.estimatedAge}j)
💚 Santé : ${healthStatus.isHealthy ? 'Saine ✅' : '⚠️ ' + healthStatus.diseaseName}

🎁 Récompenses :
  • +${rewards.coins} 🪙 pièces
  • +${rewards.xp} ⭐ XP
${rewards.bonusSeeds ? `  • +3 graines ${plantDef.emoji}` : ''}

Retrouve ton plant dans la Vue Plan du Jardin !`,
      rewards,
    };
  },

// Import
const createDigitalTwinInGarden = useGameStore((s) => (s as any).createDigitalTwinInGarden);

// State pour la position
const [twinPosition, setTwinPosition] = useState({ x: 100, y: 100 });

// Fonction
const createTwin = (photo: GardenPhoto) => {
  if (!photo.identificationResult) {
    alert('⚠️ Aucune identification.\n\nAnalysez d\'abord la photo.');
    return;
  }
  
  const plantDefId = mapPlantNameToDefId(photo.identificationResult.plantName);
  
  if (!plantDefId) {
    alert(`❌ "${photo.identificationResult.plantName}" pas dans le catalogue.

Plantes disponibles : Tomate, Poivron, Laitue, Carotte, Basilic, Fraise`);
    return;
  }
  
  // Position par défaut (coin haut-gauche du jardin)
  const result = createDigitalTwinInGarden(
    plantDefId,
    twinPosition.x,
    twinPosition.y,
    {
      plantName: photo.identificationResult.plantName,
      confidence: photo.identificationResult.confidence,
      growthStage: photo.identificationResult.growthStage,
      healthStatus: photo.identificationResult.healthStatus,
    }
  );
  
  if (result.success) {
    // Décaler la position pour le prochain plant
    setTwinPosition(prev => ({
      x: prev.x + 100,
      y: prev.y + (prev.x + 100 > 1200 ? 100 : 0), // Nouvelle ligne si déborde
    }));
  }
  
  alert(result.message);
};

// Helper
function mapPlantNameToDefId(name: string): string | null {
  const lower = name.toLowerCase();
  if (lower.includes('tomate')) return 'tomato';
  if (lower.includes('poivron') || lower.includes('piment')) return 'pepper';
  if (lower.includes('laitue')) return 'lettuce';
  if (lower.includes('carotte')) return 'carrot';
  if (lower.includes('basilic')) return 'basil';
  if (lower.includes('fraise')) return 'strawberry';
  return null;
}
