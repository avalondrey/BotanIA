/**
 * Import plants from a photo with seed rows into a mini serre
 * Maps photo coordinates to the 6x4 mini serre grid
 * @param seedRows - Array of traced rows from the photo marker
 * @param miniSerreId - ID of the target mini serre (optional, uses first if not specified)
 * @returns Success/failure object with message and stats
 */
importPlantsFromPhoto: (seedRows: Array<{
  id: string;
  color: string;
  points: { x: number; y: number }[];
  label?: string;
  plantCount?: number;
  createdAt: number;
}>, miniSerreId?: string) => {
  const state = get();
  
  // Find target mini serre
  const targetSerreId = miniSerreId || (state.miniSerres[0]?.id);
  if (!targetSerreId) {
    return {
      success: false,
      message: "❌ Aucune mini serre disponible. Veuillez d'abord acheter une mini serre dans la Boutique.",
      stats: null
    };
  }
  
  const serreIdx = state.miniSerres.findIndex((s) => s.id === targetSerreId);
  if (serreIdx < 0) {
    return {
      success: false,
      message: "❌ Mini serre introuvable.",
      stats: null
    };
  }
  
  const serre = state.miniSerres[serreIdx];
  
  // Filter out empty markers (color === 'EMPTY')
  const plantRows = seedRows.filter(r => r.color !== 'EMPTY');
  const emptyMarkers = seedRows.filter(r => r.color === 'EMPTY');
  
  if (plantRows.length === 0) {
    return {
      success: false,
      message: "❌ Aucun rang de plants trouvé. Tracez des rangs colorés sur votre photo.",
      stats: null
    };
  }
  
  // Calculate total plants needed
  let totalPlantsNeeded = 0;
  const plantsByType: Record<string, number> = {};
  
  plantRows.forEach(row => {
    const count = row.plantCount || 1;
    totalPlantsNeeded += count;
    
    // Try to extract plant type from label
    // For now, we'll ask user to select or use a default
    // In a real implementation, you'd need a dialog to map labels to plantDefIds
  });
  
  // For this implementation, we'll use a simple strategy:
  // All rows become the first available plant type the user has seeds for
  // OR return an error asking for clarification
  
  return {
    success: false,
    message: "⚠️ Fonctionnalité en développement. La correspondance label → type de plante nécessite une interface de sélection.",
    stats: {
      rowsDetected: plantRows.length,
      totalPlants: totalPlantsNeeded,
      emptySlots: emptyMarkers.length,
      available: 24,
      canFit: totalPlantsNeeded <= 24
    }
  };
},
